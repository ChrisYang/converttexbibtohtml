@inproceedings{yang2012face,
	Abstract = {In this paper, we propose a method for face parts localization called Structured-Output Regression Forests (SO-RF). We assume that the spatial graph of face parts structure can be partitioned into star graphs associated with individual parts. At each leaf, a regression model
for an individual part as well as an interdependency model between parts in the star graph is learned. During testing, individual part positions are
determined by the product of two voting maps, corresponding to two different models . The part regression model captures local feature evidence while the interdependency model captures the structure configuration. Our method has shown state of the art results on the publicly available BioID dataset and competitive results on a more challenging dataset, namely Labeled Face Parts in the Wild},
	Author = {Yang, Heng and Patras, Ioannis},

	Publisher = {Springer},
	Title = {Face Parts Localization Using Structured-output Regression Forests},
	Year = {2012}}
